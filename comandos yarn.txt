yarn init = crea un nuevo proyecto en node y crea el archivo package.json 

yarn add nombre_libreria = buscara esa libreria

yarn remove nombre_libreria = desintalara la libreria en donde sea que este ubicada

yarn global add nombre_libreria = instala una libreria de manera global

yarn | yarn install = instalare todas las librerias declaradas en el packge.json y basandose
